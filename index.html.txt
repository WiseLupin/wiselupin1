        // --- 앱 상태 ---
        let currentViewDate = new Date();
        let currentWeekStart = new Date();
        
        // --- 초기화 ---
        window.onload = function() {
            setInitialWeek();
            renderAll();
            
            // 메모 엔터 입력 이벤트
            document.getElementById('memo-text').addEventListener('keypress', function(e) {

        function renderAll() {
            renderMenuTab();
            renderPrepTab();
            loadMemos();
            loadShoppingList();
        }

        // --- 날짜 관리 ---
        function getDDay(date) {
            const diff = date.getTime() - BIRTH_DATE.getTime();
            return Math.floor(diff / (1000 * 60 * 60 * 24));
        }

        function formatDisplayDate(date) {
            const days = ['일', '월', '화', '수', '목', '금', '토'];
            return `${date.getFullYear()}.${String(date.getMonth() + 1).padStart(2, '0')}.${String(date.getDate()).padStart(2, '0')} (${days[date.getDay()]})`;
        }

        function moveDate(offset) {
            currentViewDate.setDate(currentViewDate.getDate() + offset);
            renderMenuTab();
        }

        function goToToday() {
            currentViewDate = new Date();
            renderMenuTab();
        }

        function setInitialWeek() {
            const d = new Date();
            const day = d.getDay();
            const diff = d.getDate() - day + (day === 0 ? -6 : 1); // 월요일로 설정
            currentWeekStart = new Date(d.setDate(diff));
            currentWeekStart.setHours(0, 0, 0, 0);
        }

        function goToCurrentWeek() {
            setInitialWeek();
            renderPrepTab();
        }

        // --- 탭 전환 ---
        function showTab(tab) {
            const menuTab = document.getElementById('tab-content-menu');
            const prepTab = document.getElementById('tab-content-prep');
            const menuBtn = document.getElementById('btn-menu');
            const prepBtn = document.getElementById('btn-prep');

            if (tab === 'menu') {
                menuTab.classList.remove('hidden');
                prepTab.classList.add('hidden');
                menuBtn.classList.add('tab-active');
                prepBtn.classList.remove('tab-active');
                prepBtn.classList.add('text-gray-400');
            } else {
                menuTab.classList.add('hidden');
                prepTab.classList.remove('hidden');
                prepBtn.classList.add('tab-active');
                menuBtn.classList.remove('tab-active');
                menuBtn.classList.add('text-gray-400');
                renderPrepTab();
            }
        }

        // --- 식단 카드 렌더링 ---
        function renderMenuTab() {
            const selector = document.getElementById('date-selector');
            selector.valueAsDate = new Date(currentViewDate.getTime() - currentViewDate.getTimezoneOffset() * 60000);
            
            const dDay = getDDay(currentViewDate);
            document.getElementById('d-day-info').innerText = `D+${dDay} | ${formatDisplayDate(currentViewDate)}`;

            const list = document.getElementById('meals-list');
            list.innerHTML = "";

            const dayMeal = mealData[dDay];
            if (!dayMeal) {
                list.innerHTML = `<div class="text-center py-10 text-gray-300">식단 데이터가 없습니다.<br><small>(D+258 ~ D+373 범위 내에만 있습니다)</small></div>`;
                return;
            }

            const sessions = [
                { name: '아침 ☀️', content: dayMeal.morning },
                { name: '점심 ☁️', content: dayMeal.lunch },
                { name: '저녁 🌙', content: dayMeal.dinner }
            ];

            sessions.forEach(s => {
                if (!s.content) return;
                const card = document.createElement('div');
                card.className = "bg-orange-50 border-l-4 border-orange-300 p-4 rounded-r-2xl";
                card.innerHTML = `<h4 class="font-bold text-orange-600 mb-2 text-sm">${s.name}</h4>`;
                
                const contentWrapper = document.createElement('div');
                contentWrapper.className = "space-y-3";

                // 베이스와 토핑 분리 처리
                const basePart = s.content.match(/베이스:\s*([^/|]*)/);
                const toppingPart = s.content.match(/토핑:\s*(.*)/);

                if (basePart) {
                    const baseDiv = document.createElement('div');
                    baseDiv.innerHTML = `<div class="text-[10px] text-gray-400 mb-1">BASE</div>`;
                    const tag = document.createElement('span');
                    tag.className = "ingredient-tag tag-base";
                    tag.innerText = basePart[1].trim();
                    baseDiv.appendChild(tag);
                    contentWrapper.appendChild(baseDiv);
                }

                if (toppingPart) {
                    const toppingDiv = document.createElement('div');
                    toppingDiv.innerHTML = `<div class="text-[10px] text-gray-400 mb-1">TOPPING</div>`;
                    const tagsContainer = document.createElement('div');
                    tagsContainer.className = "flex flex-wrap gap-0.5";
                    
                    const items = toppingPart[1].split(/[,/]/);
                    items.forEach(item => {
                        const cleanItem = item.trim();
                        if (!cleanItem) return;
                        const span = document.createElement('span');
                        span.className = `ingredient-tag ${getColorClass(cleanItem)}`;
                        span.innerText = cleanItem;
                        tagsContainer.appendChild(span);
                    });
                    toppingDiv.appendChild(tagsContainer);
                    contentWrapper.appendChild(toppingDiv);
                }

                // 베이스/토핑 형식이 아닌 경우 (완료기 등)
                if (!basePart && !toppingPart) {
                    const tagsContainer = document.createElement('div');
                    tagsContainer.className = "flex flex-wrap gap-1";
                    const items = s.content.split(/[,/]/);
                    items.forEach(item => {
                        const cleanItem = item.trim();
                        if (!cleanItem) return;
                        const span = document.createElement('span');
                        span.className = `ingredient-tag ${getColorClass(cleanItem)}`;
                        span.innerText = cleanItem;
                        tagsContainer.appendChild(span);
                    });
                    contentWrapper.appendChild(tagsContainer);
                }

                card.appendChild(contentWrapper);
                list.appendChild(card);
            });
        }

        function getColorClass(text) {
            if (text.includes('소고기')) return 'tag-beef';
            if (text.includes('닭고기')) return 'tag-chicken';
            if (text.includes('생선') || text.includes('오징어') || text.includes('새우') || text.includes('게살') || text.includes('어기')) return 'tag-fish';
            if (text.includes('시금치') || text.includes('브로콜리') || text.includes('아욱') || text.includes('청경채') || text.includes('비타민') || text.includes('케일') || text.includes('쑥갓')) return 'tag-veg-green';
            if (text.includes('당근') || text.includes('토마토') || text.includes('파프리카')) return 'tag-veg-red';
            if (text.includes('비트')) return 'tag-beet';
            if (text.includes('단호박')) return 'tag-pumpkin';
            if (text.includes('달걀') || text.includes('계란')) return 'tag-egg';
            return 'tag-default';
        }

        // --- 메모 기능 ---
        function toggleMemoInput() {
            const area = document.getElementById('memo-input-area');
            area.classList.toggle('hidden');
            if (!area.classList.contains('hidden')) document.getElementById('memo-text').focus();
        }

        function saveMemo() {
            const input = document.getElementById('memo-text');
            const text = input.value.trim();
            if (!text) return;

            const memos = JSON.parse(localStorage.getItem('dain_memos') || "[]");
            memos.push(text);
            localStorage.setItem('dain_memos', JSON.stringify(memos));

            input.value = "";
            toggleMemoInput();
            loadMemos();
        }

        function loadMemos() {
            const list = document.getElementById('memo-list');
            list.innerHTML = "";
            const memos = JSON.parse(localStorage.getItem('dain_memos') || "[]");

            memos.forEach((memo, index) => {
                const li = document.createElement('li');
                li.className = "flex justify-between items-center bg-white border border-gray-100 p-3 rounded-xl shadow-sm text-sm";
                li.innerHTML = `
                    <span>• ${memo}</span>
                    <button onclick="deleteMemo(${index})" class="text-red-300 ml-2">✕</button>
                `;
                list.appendChild(li);
            });
        }

        function deleteMemo(index) {
            const memos = JSON.parse(localStorage.getItem('dain_memos') || "[]");
            memos.splice(index, 1);
            localStorage.setItem('dain_memos', JSON.stringify(memos));
            loadMemos();
        }

        // --- 준비 탭 (주간 재료) ---
        function renderPrepTab() {
            const end = new Date(currentWeekStart);
            end.setDate(end.getDate() + 6);
            document.getElementById('week-range').innerText = `${formatDateShort(currentWeekStart)} ~ ${formatDateShort(end)}`;

            const ingredients = new Set();
            for (let i = 0; i < 7; i++) {
                const d = new Date(currentWeekStart);
                d.setDate(d.getDate() + i);
                const meal = mealData[getDDay(d)];
                if (meal) {
                    [meal.morning, meal.lunch, meal.dinner].forEach(m => {
                        if (!m) return;
                        const items = m.split(/[,/]/);
                        items.forEach(it => {
                            const clean = it.trim().replace(/베이스:|토핑:|1\.|2\./g, "").trim();
                            if(clean && clean.length > 1) ingredients.add(clean);
                        });
                    });
                }
            }

            const container = document.getElementById('available-ingredients');
            container.innerHTML = "";
            ingredients.forEach(ing => {
                const btn = document.createElement('button');
                btn.className = "px-3 py-1 bg-gray-100 rounded-full text-xs text-gray-600 hover:bg-blue-100";
                btn.innerText = ing;
                btn.onclick = () => addToShoppingList(ing);
                container.appendChild(btn);
            });
        }

        function moveWeek(offset) {
            currentWeekStart.setDate(currentWeekStart.getDate() + (offset * 7));
            renderPrepTab();
        }

        function formatDateShort(date) {
            return `${date.getMonth() + 1}.${date.getDate()}`;
        }

        function addToShoppingList(item) {
            const list = JSON.parse(localStorage.getItem('dain_shopping') || "[]");
            if (!list.includes(item)) {
                list.push(item);
                localStorage.setItem('dain_shopping', JSON.stringify(list));
                loadShoppingList();
            }
        }

        function loadShoppingList() {
            const container = document.getElementById('shopping-list');
            container.innerHTML = "";
            const list = JSON.parse(localStorage.getItem('dain_shopping') || "[]");

            list.forEach((item, index) => {
                const div = document.createElement('div');
                div.className = "flex justify-between items-center bg-white p-2 rounded-lg text-sm border-l-4 border-blue-300";
                div.innerHTML = `<span>${item}</span><button onclick="removeFromShopping(${index})" class="text-gray-300">✕</button>`;
                container.appendChild(div);
            });
        }

        function removeFromShopping(index) {
            const list = JSON.parse(localStorage.getItem('dain_shopping') || "[]");
            list.splice(index, 1);
            localStorage.setItem('dain_shopping', JSON.stringify(list));
            loadShoppingList();
        }

            currentViewDate = new Date(this.value);
